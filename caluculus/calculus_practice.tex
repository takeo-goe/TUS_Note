% ===== 例題1 =====
\begin{itembox}[l]{例題1}
$x,y,z$平面および下の図の平面に囲まれた領域の体積を求めよ。

\vspace{10pt}

\begin{center}
\includegraphics[width=0.5\linewidth]{images/number1.png}
\end{center}

\end{itembox}



\begin{center}
\includegraphics[width=0.8\linewidth]{images/number2.png}
\end{center}

\vspace{10pt}
\fbox{方針}\par



①平面を$z = f(x,y)$の形式で明示する.\par

②積分範囲を$D = \{(x,y)|xとyの条件\}$という形式で明示する.\par

③二重積分を立式する.

\vspace{10pt}
\fbox{解答}\par
平面は
\begin{align*}
    z = 1-x-y
\end{align*}
とかける. 指定の領域の体積を計算するには, 積分範囲を
\begin{align*}
    D = \{(x,y)|0 \le x \le 1, 0 \le y \le 1-x\}
\end{align*}
として二重積分を行う.

\begin{align*}
    V &= \iint_{D}^{} z dx dy\\
      &= \int_{0}^{1} (\int_{0}^{1-x}(1-x-y)dy)dx  \\
      &=...  
\end{align*}



% === 例題2 ===
\begin{itembox}[l]{例題2}
\begin{align*}
    \iint_{D}^{} (3x + 2y) dx dy, D = \{(x,y) | 0 \le x \le 1, 0 \le y \le 2\}
\end{align*}
を累次積分で求めよ.
\end{itembox}

\fbox{解答}
\begin{align*}
    V &= \int_{0}^{2} (\int_{0}^{1} (3x + 2y)dx)dy\\
      &= \int_{0}^{2} \left[\frac{3}{2} x^2 + 2yx \right]_{0}^{1}\\
      &= \int_{0}^{2} (\frac{3}{2} + 2y)dy\\
      &= \left[\frac{3}{2}y + y^2 \right]_{0}^{2}.
\end{align*}





% === 例題3 ===
\begin{itembox}[l]{例題3}
\begin{align*}
    V = \iint_{D}^{} x y^2 dx dy, D = \{(x,y) | 0 \le x \le 1, 0 \le y \le 2\}
\end{align*}
を変数分離によって求めよ.
\end{itembox}

\fbox{解答}
\begin{align*}
    V &= \int_{0}^{1} x\, dx \int_{0}^{2} y^2\, dy\\
      &= \left[ \frac{1}{2} x^2 \right]_{0}^{1}
         \left[ \frac{1}{3} y^3 \right]_{0}^{2}\\
      &= \frac{4}{3}.
\end{align*}
